import random # Function to display the current score def display_score(score): print(f"Your current score: {score}") # Function to play the game def play_game(score): print("\nWelcome to Soccer Hut!") print("Press 'p' to play or 'q' to quit.") while True: choice = input("Enter your choice: ").strip().lower() if choice == 'p': result = random.choice(['goal', 'miss']) if result == 'goal': points = random.randint(1, 10) score += points print(f"Goal! You scored {points} points.") else: print("Miss! No points scored.") elif choice == 'q': break else: print("Invalid choice. Please enter 'p' to play or 'q' to quit.") display_score(score) return score # Main function def main(): score = 0 print("Welcome to Soccer Hut!") score = play_game(score) print(f"\nGame Over! Your final score is {score}.") if __name__ == "__main__": main()
