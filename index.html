from flask import Flask, render_template, request, redirect, url_for app = Flask(__name__) # Store dishes in a dictionary with a password for each dish dishes = {} @app.route('/') def index(): return render_template('index.html', dishes=dishes) @app.route('/add_dish', methods=['POST']) def add_dish(): dish_name = request.form['dish_name'] price = request.form['price'] password = request.form['password'] if dish_name and price and password: # Add dish to the dictionary with the password dishes[dish_name] = {'price': price, 'password': password} return redirect(url_for('index')) @app.route('/remove_dish', methods=['POST']) def remove_dish(): dish_name = request.form['dish_name'] password = request.form['password'] # Check if the dish exists and the password matches if dish_name in dishes and dishes[dish_name]['password'] == password: del dishes[dish_name] return redirect(url_for('index')) if __name__ == '__main__': app.run(debug=True)
